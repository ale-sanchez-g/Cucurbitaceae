# filename: grid.rb

require 'selenium-webdriver'
require 'rspec/expectations'
include RSpec::Matchers

def setup
  @driver = Selenium::WebDriver.for(
    :remote,
    url: 'http://localhost:4400/wd/hub',
    desired_capabilities: :firefox) # you can also use :chrome, :safari, etc.
end

def teardown
  @driver.quit
end

def run
  setup
  yield
  teardown
end

run do	
  @driver.navigate.to "http://172.20.10.7/reports/index.php"
  puts @driver.title
  idSession = @driver.instance_variable_get('@bridge').instance_variable_get('@session_id')
  puts idSession
  @driver.navigate.to "http://172.20.10.7:80/reports/insert.php?id=#{idSession}&name=alejandrogrid&status=0&tags="
end
